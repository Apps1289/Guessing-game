# Use an official Maven image to build the app
FROM maven:3.8.7-openjdk-17 AS build

# Set working directory inside the container
WORKDIR /app

# Copy your app's code into the container
COPY . .

# Build the project and package it as a JAR
RUN mvn clean package -DskipTests

# Use a slim OpenJDK image for running the app
FROM openjdk:17-jdk-slim

WORKDIR /app

# Copy the built JAR from the previous image
COPY --from=build /app/target/guessing-game-1.0.0.jar .

# Set the command to run the JAR
CMD ["java", "-jar", "guessing-game-1.0.0.jar"]
